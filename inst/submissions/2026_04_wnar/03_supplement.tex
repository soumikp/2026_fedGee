% The widespread adoption of Electronic Health Record (EHR) systems has enabled large-scale multicenter studies. However, data privacy protection restricts the sharing of patient-level data, creating barriers to pooled analysis. Federated learning (FL) provides a solution by enabling model estimation through the exchange of summary statistics rather than raw data. Standard FL methods require independent and identically distributed data, which is unrealistic in EHR systems that capture longitudinal data. We propose a FL framework using Generalized Estimating Equations (Fed-GEE) for EHR data that enables privacy-preserving inference for longitudinal EHR data. Extensive simulation studies comparing FedGEE with competing FL models reveal improvements in terms of reduced bias, improved coverage probability, and reduced runtimes. We investigate key asymptotic properties of FedGEE, providing a theoretical basis for valid population-level inference. We demonstrate the performance of FedGEE in numerical experiments and implement an application to longitudinal EHR data from the Veterans Health Administration, the largest integrated EHR system in the United States.

\section{Supplement A: Algorithm for inference and estimation for Fed-GEE}\label{app:algo}
\begin{algorithm}
\footnotesize
\label{alg:fedgee}
\begin{algorithmic}[1]
\REQUIRE 
    Distributed datasets $\{\mathbf{X}_i, \mathbf{Y}_i\}_{i=1}^K$ across $K$ sites; 
    Initial parameter vector $\boldsymbol{\beta}^{(0)}$ (with site-level covariates initialized to 0); 
    Working correlation parameter $\alpha$; 
    Convergence tolerance $\epsilon$; 
    Maximum iterations $M$.
\ENSURE 
    Estimated marginal parameters $\hat{\boldsymbol{\beta}}$; 
    KC-corrected robust sandwich variance $\mathbf{V}_{\beta}^{KC}$.

\STATE \textbf{Phase 1: Federated Fisher Scoring: Iterative optimization}
\FOR{$m = 0, 1, \dots, M-1$}
    \STATE Server broadcasts global parameters $\boldsymbol{\beta}^{(m)}$ to all $K$ sites.
    \FOR{\textbf{each site} $i \in \{1, \dots, K\}$ \textbf{in parallel}}
        \STATE \textit{// Execute locally without exposing patient data}
        \STATE Compute local Score vector: $\mathbf{S}_i\left(\boldsymbol{\beta}^{(m)}\right) = \sum_{j=1}^{n_i} \mathbf{D}_{ij}^T \mathbf{V}_{ij}^{-1} \left(\mathbf{Y}_{ij} - \boldsymbol{\mu}_{ij}\right)$
        \STATE Compute local Bread matrix: $\mathbf{B}_i\left(\boldsymbol{\beta}^{(m)}\right) = \sum_{j=1}^{n_i} \mathbf{D}_{ij}^T \mathbf{V}_{ij}^{-1} \mathbf{D}_{ij}$
        \STATE Transmit $\{\mathbf{S}_i, \mathbf{B}_i\}$ to the central server.
    \ENDFOR
    
    \STATE \textbf{Server Execution:}
    \STATE Aggregate global matrices: $\mathbf{S}_{global} = \sum_{i=1}^K \mathbf{S}_i$ and $\mathbf{B}_{global} = \sum_{i=1}^K \mathbf{B}_i$
    \STATE Update parameters: $\boldsymbol{\beta}^{(m+1)} = \boldsymbol{\beta}^{(m)} + \mathbf{B}_{global}^{-1} \mathbf{S}_{global}$
    \IF{$\left|\left|\boldsymbol{\beta}^{(m+1)} - \boldsymbol{\beta}^{(m)}\right|\right| < \epsilon$}
        \STATE Set $\hat{\boldsymbol{\beta}} \leftarrow \boldsymbol{\beta}^{(m+1)}$ and \textbf{break}
    \ENDIF
\ENDFOR

\STATE \textbf{Phase 2: Kauermann-Carroll Small-Sample Variance Estimation}
\STATE Server broadcasts final parameters $\hat{\boldsymbol{\beta}}$ and final global Bread $\mathbf{B}_{global}$ to all $K$ sites.

\FOR{\textbf{each site} $i \in \{1, \dots, K\}$ \textbf{in parallel}}
    \STATE Compute final local Score evaluated at convergence: $\mathbf{S}_i\left(\hat{\boldsymbol{\beta}}\right)$
    \STATE Compute local Bread evaluated at convergence: $\mathbf{B}_i\left(\hat{\boldsymbol{\beta}}\right)$
    \STATE Calculate local site leverage: $\mathbf{H}_{ii} = \mathbf{B}_i \mathbf{B}_{global}^{-1}$
    \STATE Apply leverage to adjust score: $\widetilde{\mathbf{S}}_i = \left(\mathbf{I} - \mathbf{H}_{ii}\right)^{-1/2} \mathbf{S}_i\left(\hat{\boldsymbol{\beta}}\right)$
    \STATE Construct bias-corrected local Meat matrix: $\mathbf{M}_i^{KC} = \widetilde{\mathbf{S}}_i \widetilde{\mathbf{S}}_i^T$
    \STATE Transmit $\mathbf{M}_i^{KC}$ to the central server.
\ENDFOR

\STATE \textbf{Server Execution:}
\STATE Aggregate global corrected Meat matrix: $\mathbf{M}_{global}^{KC} = \sum_{i=1}^K \mathbf{M}_i^{KC}$
\STATE Construct final robust variance: $\mathbf{V}_{\beta}^{KC} = \mathbf{B}_{global}^{-1} \mathbf{M}_{global}^{KC} \mathbf{B}_{global}^{-1}$

\RETURN $\hat{\boldsymbol{\beta}}$ and $\mathbf{V}_{\beta}^{KC}$
\end{algorithmic}
\end{algorithm}

\section{Supplement B: Proof of Theorem 1} \label{a:proof_t1}

\begin{proof}
Let $\mathbf{S}_{cen}(\boldsymbol{\beta})$ and $\mathbf{B}_{cen}(\boldsymbol{\beta})$ represent the score vector and information (bread) matrix for the centralized dataset. By the additive property of matrix operations over independent subjects across sites, we can decompose the centralized matrices:
\begin{equation}
    \mathbf{S}_{cen}(\boldsymbol{\beta}) = \sum_{i=1}^K \sum_{j=1}^{n_i} \mathbf{D}_{ij}^T \mathbf{V}_{ij}^{-1} (\mathbf{Y}_{ij} - \boldsymbol{\mu}_{ij}) = \sum_{i=1}^K \mathbf{S}_i(\boldsymbol{\beta}) = \mathbf{S}_{global}(\boldsymbol{\beta})
\end{equation}
\begin{equation}
    \mathbf{B}_{cen}(\boldsymbol{\beta}) = \sum_{i=1}^K \sum_{j=1}^{n_i} \mathbf{D}_{ij}^T \mathbf{V}_{ij}^{-1} \mathbf{D}_{ij} = \sum_{i=1}^K \mathbf{B}_i(\boldsymbol{\beta}) = \mathbf{B}_{global}(\boldsymbol{\beta})
\end{equation}
At any iteration $m$, assuming $\boldsymbol{\beta}^{(m)}_{fed} = \boldsymbol{\beta}^{(m)}_{cen}$, the centralized Newton-Raphson update is $\boldsymbol{\beta}^{(m+1)}_{cen} = \boldsymbol{\beta}^{(m)}_{cen} + \mathbf{B}_{cen}^{-1} \mathbf{S}_{cen}$. Substituting the sums yields exactly the Fed-GEE server update rule defined in Equation 1. By induction, the iterative sequences are identical, yielding $\hat{\boldsymbol{\beta}}_{fed} = \hat{\boldsymbol{\beta}}_{cen}$.
\end{proof}


\section{Supplement C: Proof of Theorem 2}
\begin{proof}
By Theorem  1, $\hat{\boldsymbol{\beta}}_{fed}$ is the root of the global score function such that $\mathbf{S}_{global}(\hat{\boldsymbol{\beta}}_{fed}) = \mathbf{0}$. We take a first-order Taylor series expansion of $\mathbf{S}_{global}(\hat{\boldsymbol{\beta}}_{fed})$ around the true $\boldsymbol{\beta}_0$:
\begin{equation}
    \mathbf{0} = \mathbf{S}_{global}\left(\boldsymbol{\beta}_0\right) + \left[ \frac{\partial \mathbf{S}_{global}\left(\boldsymbol{\beta}^*\right)}{\partial \boldsymbol{\beta}^T} \right] \left(\hat{\boldsymbol{\beta}}_{fed} - \boldsymbol{\beta}_0\right)
\end{equation}
where $\boldsymbol{\beta}^*$ lies on the line segment between $\hat{\boldsymbol{\beta}}_{fed}$ and $\boldsymbol{\beta}_0$. Note that for generalized estimating equations, the negative derivative of the score function is approximated by the bread matrix, $-\frac{\partial \mathbf{S}_{global}(\boldsymbol{\beta})}{\partial \boldsymbol{\beta}^T} \approx \mathbf{B}_{global}(\boldsymbol{\beta})$. Rearranging the terms and multiplying by $\sqrt{K} / K$, we obtain:
\begin{equation}
    \sqrt{K}\left(\hat{\boldsymbol{\beta}}_{fed} - \boldsymbol{\beta}_0\right) = \left( \frac{1}{K} \mathbf{B}_{global}(\boldsymbol{\beta}^*) \right)^{-1} \left( \frac{1}{\sqrt{K}} \mathbf{S}_{global}(\boldsymbol{\beta}_0) \right)
\end{equation}

We analyze the two components on the right-hand side separately. First, because the marginal mean is correctly specified (assumption 1), we have $E[\mathbf{S}_i(\boldsymbol{\beta}_0)] = \mathbf{0}$. Furthermore, bounded cluster sizes (assumption 2) and bounded covariates (assumption 3) ensure that the variance of each independent site-level score vector, is strictly finite. The global score evaluated at $\boldsymbol{\beta}_0$ is a sum of $K$ independent, zero-mean random vectors. By the multivariate central limit theorem, we have:
\begin{equation}
    \frac{1}{\sqrt{K}} \mathbf{S}_{global}(\boldsymbol{\beta}_0) = \frac{1}{\sqrt{K}} \sum_{i=1}^K \mathbf{S}_i(\boldsymbol{\beta}_0) \xrightarrow{d} \mathcal{N}(\mathbf{0}, \mathcal{M}(\boldsymbol{\beta}_0))
\end{equation}
Second, the application of the weak law of large numbers requires the expected value of the local bread matrices to be bounded. Because the local bread matrix is a function of the covariates and is summed over the cluster size $K$ assumption 2 and 3  guarantee that the elements remain finite. Consequently, under assumption 4, the weak law of large numbers implies that $\frac{1}{K} \mathbf{B}_{global}(\boldsymbol{\beta}^*) \xrightarrow{p} \mathcal{B}(\boldsymbol{\beta}_0)$. Since $\mathcal{B}(\boldsymbol{\beta}_0)$ is invertible, by the continuous mapping theorem, the inverse converges in probability to $\mathcal{B}^{-1}(\boldsymbol{\beta}_0)$. Combining these results using Slutsky's theorem, we establish asymptotic normality:
\begin{equation}
    \sqrt{K}(\hat{\boldsymbol{\beta}}_{fed} - \boldsymbol{\beta}_0) \xrightarrow{d} \mathcal{B}^{-1}(\boldsymbol{\beta}_0) \cdot \mathcal{N}(\mathbf{0}, \mathcal{M}(\boldsymbol{\beta}_0)) \equiv \mathcal{N}\left(\mathbf{0}, \mathcal{B}^{-1} \mathcal{M} \mathcal{B}^{-1}\right)
\end{equation}
Consistency (i.e., $\hat{\boldsymbol{\beta}}_{fed} \xrightarrow{p} \boldsymbol{\beta}_0$) follows immediately from this asymptotic normality. 

Finally, for the Kauermann-Carroll adjusted meat matrix $\mathbf{M}_{global}^{KC}$, observe that the local leverage matrices are defined as $\mathbf{H}_{ii} = \mathbf{B}_i \left(\sum_i \mathbf{B}_i\right)^{-1}$. As $K \to \infty$, the global information matrix grows unboundedly, pushing the leverage terms $\mathbf{H}_{ii} \to \mathbf{0}$. Consequently, the adjustment term $(\mathbf{I} - \mathbf{H}_{ii})^{-1/2} \to \mathbf{I}$, demonstrating that the KC correction vanishes asymptotically, yielding the standard consistent sandwich estimator while providing necessary Type I error control in finite-sample distributed settings.
\end{proof}

\begin{remark}[Score-Level Leverage Correction]
\label{rem:score_md}
We note that the classical Kauermann-Carroll correction operates on the observation-level residuals via the $N_i \times N_i$ hat matrix $\mathbf{H}_i = \mathbf{D}_i \mathbf{B}^{-1} \mathbf{D}_i^T \mathbf{V}_i^{-1}$. In our federated setting, transmitting observation-level adjusted residuals would compromise the privacy guarantees of the protocol. We therefore apply an analogous correction in the score space, replacing the observation-level leverage with the $p \times p$ score-level leverage $\mathbf{H}_{ii}^{(s)} = \mathbf{B}_i \mathbf{B}^{-1}$.
\end{remark}
